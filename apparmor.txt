# AppArmor profile for the Familienkalender add-on.
#
# File name and location are fixed by the Supervisor contract: an
# `apparmor.txt` next to config.yaml is picked up automatically when the
# add-on is (re)installed; the default `apparmor: true` then uses this
# custom profile instead of the generic docker-default one. The profile
# must contain exactly one top-level profile whose name is the add-on
# slug from config.yaml — the Supervisor rewrites it to the installed
# slug (e.g. local_familienkalender) via adjust_profile() before loading.
#
# Structure follows the official HA example profile (docs: add-on
# security) — a fully enumerated allowlist broke the s6-overlay init
# chain on the device in ways that are hard to debug without host dmesg
# access, so we use the template's broad `file,` grant instead. The
# effective confinement comes from the explicit deny rules (deny beats
# allow in AppArmor): no access to other add-ons, HA config, backups or
# shares, no writes to the SSL certs (read-only for the TLS feed
# listener), no mounts, no ptrace. Capabilities and network are still
# enumerated narrowly.

#include <tunables/global>

profile familienkalender flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Explicit denials: these hold even against the broad `file,` below.
  deny /addons/{,**} rwklx,
  deny /config/{,**} rwklx,
  deny /backup/{,**} rwklx,
  deny /share/{,**} rwklx,
  deny /media/{,**} rwklx,
  deny mount,
  deny umount,
  deny ptrace,

  # Broad file access as in the official template; execution rights are
  # granted separately below (file, does not imply x).
  file,

  # Capabilities: run.sh chowns /data to the app user and drops
  # privileges via su-exec (setuid/setgid).
  capability chown,
  capability fowner,
  capability fsetid,
  capability dac_override,
  capability dac_read_search,
  capability setuid,
  capability setgid,
  capability net_bind_service,

  # Signals: s6/bashio send lifecycle signals; uvicorn receives them.
  signal (send) set=(kill,term,int,hup,cont),
  signal (receive),

  # Network: uvicorn listens on 8099; sync fetches calendars over
  # HTTPS (inet/inet6 stream) and resolves names via DNS (dgram).
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  network unix stream,
  network netlink raw,

  # S6-Overlay init chain (official template)
  /init rix,
  /bin/** rix,
  /sbin/** rix,
  /usr/bin/** rix,
  /usr/sbin/** rix,
  /run/{s6,s6-rc*,service}/** rwix,
  /package/** rix,
  /command/** rix,
  /etc/s6-overlay/{,**} rix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio (sourced by run.sh)
  /usr/lib/bashio/** rix,
  /tmp/{,**} rwk,

  # Start script and Python (HA base-python installs to /usr/local).
  /run.sh rix,
  /usr/local/bin/** rix,

  # Persistent data: SQLite needs fcntl locks (k) on the database and
  # its journal/WAL siblings; token files live here too.
  /data/ rw,
  /data/** rwk,

  # HA certificates for the TLS feed listener: strictly read-only. The
  # broad `file,` above would also grant write, so writes (and lock/exec)
  # are denied explicitly — the mount is ro (map: ssl:ro) as well, this
  # is defense in depth.
  deny /ssl/{,**} wklx,
  /ssl/ r,
  /ssl/** r,
}
